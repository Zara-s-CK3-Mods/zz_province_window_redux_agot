zz_valyria_province ={
	scope = province 	
	is_valid = {
		scope:player = {
			gold >= 1500
		}
		OR = {
			county.holder = scope:player
			county.holder.top_liege = scope:player
		}
		
	}
	is_shown = {
		not = {
			has_holding_type = ruin_holding
		}
	}
	effect = {
		add_province_modifier = extra_building_slot
		scope:player = {
			remove_short_term_gold = {
				gold = {
					value = 1500
				}
			}
		}
	}
}

zz_valyria_province_holding_list = {
	scope = character 
	effect = {
		clear_variable_list = zz_valyria_holding_list
		trait:holding_type_01 = {
			root = {
				add_to_variable_list = {
					name = zz_valyria_holding_list
					target = prev
				}
			}
		}
		trait:holding_type_02 = {
			root = {
				add_to_variable_list = {
					name = zz_valyria_holding_list
					target = prev
				}
			}
		}
		trait:holding_type_03 = {
			root = {
				add_to_variable_list = {
					name = zz_valyria_holding_list
					target = prev
				}
			}
		}
	}
}
zz_valyria_swap_holding_type ={
	scope = province 	
	saved_scopes = {
		holding_type 
		player
	}
	is_valid = {
		scope:player = {
			gold >= 2500
		}
	}

	effect = {
		if = { 
			limit = { 
				scope:holding_type ?=  trait:holding_type_03 
			}
			set_holding_type = city_holding
		}
		if = { 
			limit = { 
				scope:holding_type ?= trait:holding_type_02
			}
			set_holding_type = church_holding
		}
		if = { 
			limit = { 
				scope:holding_type ?= trait:holding_type_01
			}
			set_holding_type = castle_holding
		}
		scope:player = {
			remove_short_term_gold = {
				gold = {
					value = 2500
				}
			}
		}
	}
}

		}
	}

zz_agot_is_unrestored = {
	scope = title
	is_shown = {
		global_var:zz_acm = 1
		has_holding_type = ruin_holding
		NOT = {
			county ?= {
				holder ?= { has_government = ruins_government }
			}
		}
	}
}

zz_valyria_dragon_visit = {
	scope = province 	
	is_shown = {
		scope:actor = { 
			has_trait = dragonrider
		}
	}
	is_valid = { 
		scope:actor = { 
			is_available = yes
		}
	}
	effect = {
		scope:actor = {
			start_travel_plan = {
				destination = ROOT
				on_arrival_event = zz_valyria_province.0001
				on_arrival_destinations = all_but_last
			}
		}
		
	}
}

zz_valyria_province_cast_click = {
	scope = character
	is_shown = {
		zz_valyria_is_mage = yes 
	}
	is_valid = {

	}
	effect = {
		save_scope_as = actor 

		set_variable = {
			name = zz_selected_mystery_province 
			value = scope:selected_province
		}
		set_variable = {
			name = zz_selected_mystery_county 
			value = scope:selected_province.county
		}
		scope:selected_province = {
			province_owner = {
				save_scope_as = recipient
			}
		}
		zz_ui_setup_effect = yes 
	}
}
